# Custom Suricata Rules - /var/lib/suricata/rules/local.rules
# Lab 3: IDS Detection Rules

# Rule 1: Detect ICMP ping traffic to HOME_NET
alert icmp any any -> $HOME_NET any (msg:"ICMP Ping Detected"; sid:1000001; rev:1;)

# Rule 2: Detect TCP port scan (SYN packets to multiple ports)
alert tcp any any -> $HOME_NET any (msg:"Possible TCP Port Scan"; flags:S; threshold:type threshold, track by_src, count 5, seconds 60; sid:1000002; rev:1;)
